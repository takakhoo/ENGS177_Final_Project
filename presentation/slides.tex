% --- 3. The problem ---
\begin{frame}{Why static allocation breaks}
The 60/40 portfolio assumes a stable joint return distribution. Reality violates this:
\bigskip

\begin{itemize}\setlength{\itemsep}{6pt}
  \item \alert{Conditional heteroskedasticity}: calm decades broken by short crises (2008, 2020, 2022).
  \item \alert{Stress correlation}: in crises, stocks and bonds sell off \emph{together}. Diversification erodes exactly when it's most needed.
\end{itemize}

\bigskip\bigskip
\begin{center}
\begin{tikzpicture}
\node[hook,text width=0.82\textwidth] {\centering
  \large \textbf{Question.} Can a fully decision-theoretic regime overlay\\
  beat the static 60/40 after realistic transaction costs?
};
\end{tikzpicture}
\end{center}
\end{frame}

% --- 4. Why beyond money ---
\begin{frame}{Why we're actually doing this}
Money is the scoreboard. The \emph{methods question} is the point:

\bigskip
\begin{itemize}\setlength{\itemsep}{6pt}
  \item Test the implicit claim of every regime-switching paper.
  \item Build the policy from \alert{explicit optimisation}, not heuristic.
  \item Treat the regime as a \alert{hidden state} inferred by Bayes.
  \item Make the policy \alert{falsifiable} (POMDP+QMDP).
\end{itemize}

\bigskip
\begin{center}
\textit{If we don't beat 60/40, we want to know \textbf{exactly which methodological choice} drives the failure.}

% --- 6. Analogy 1: doctor ---
\begin{frame}{Analogy 1: the doctor with symptoms (= POMDP)}
\begin{columns}[T]
\begin{column}{0.58\textwidth}
You are a doctor. Your patient is the market.

\bigskip
\begin{itemize}\setlength{\itemsep}{4pt}

  \item You only see symptoms (VIX, term spread).
  \item Different treatments (portfolio weights) work better in different states.
  \item Goal: \alert{read the symptoms, infer the state, prescribe}.
\end{itemize}
\end{column}

  \draw[->] (o) -- (a);
\end{tikzpicture}

\bigskip\smallskip
\footnotesize\textit{This IS a POMDP.}\\
\textit{$s_t$ = regime, $\mathbf{o}_t$ = obs,}\\
\textit{$\mathbf{a}_t$ = weights.}
\end{column}
\end{columns}

% --- 7. Analogy 2: casino dealer ---
\begin{frame}{Analogy 2: the dealer secretly switching decks (= HMM)}
A poker dealer secretly swaps between two decks between rounds. You never see which deck. You only see the cards.
\bigskip

\begin{itemize}\setlength{\itemsep}{4pt}
  \item \alert{Transition matrix $T$:} how often the dealer switches. We estimate $\hat T \approx \begin{psmallmatrix}0.96 & 0.04\\ 0.06 & 0.94\end{psmallmatrix}$ (sticky: rarely switches).
  \item \alert{Emission $O(\mathbf{o}\mid s)$:} the card distribution under each deck. Gaussian: $\mathcal{N}(\boldsymbol\mu_s, \boldsymbol\Sigma_s)$.
  \item \alert{Baum--Welch (EM):} from a long stream of cards, figure out each deck and the switch rate.
\end{itemize}

\bigskip
\centering\textit{The Gaussian HMM \emph{is} the formal model of this dealer.}
\end{frame}

% --- 8. Analogy 3: Bayes filter ---
\begin{frame}{Analogy 3: ``which deck is in play \emph{right now}?'' (= Bayes filter)}
After each card, you update your guess.
\bigskip

\textbf{Predict:} ``The dealer might have switched.''\\[2pt]
$\quad\hat b_t(s') = \sum_s T(s'\mid s)\,b_{t-1}(s).$

\medskip
\textbf{Update:} ``A high card just came out, so it's much more likely Deck A.''\\[2pt]
$\quad b_t(s') \propto O(\mathbf{o}_t\mid s')\cdot\hat b_t(s').$

\bigskip
\begin{tikzpicture}
\node[winner,text width=0.86\textwidth]{\centering
  \large One observation of VIX$=35$ flips belief\\
  from \textbf{70/30 bull} $\to$ \textbf{93/7 bear} in one step.\\
  \smallskip\small That responsiveness is the whole point of the filter.
};
\end{tikzpicture}
\end{frame}

% --- 9. Analogy 4: MDP ---
\begin{frame}{Analogy 4: ``what if the dealer dealt face-up?'' (= MDP)}
Suppose the dealer turned the decks face-up, so you can see which deck is in use at every moment.

\bigskip
What's the best action in each deck? Solve the Bellman equation:
\[
V^\ast(s) = \max_{\mathbf{a}}\Bigl\{R(s,\mathbf{a}) + \lambda\sum_{s'} T(s'\mid s)V^\ast(s')\Bigr\}
\]

\bigskip
The answer is a table:
\begin{center}\small
\begin{tabular}{ll}
``In Deck A (bull regime):'' & do X. \\
``In Deck B (bear regime):'' & do Y. \\
\end{tabular}
\end{center}

\bigskip
\centering\textit{This is the easier sub-problem. We solve it by value iteration AND policy iteration.}
\end{frame}

% --- 10. Analogy 5: QMDP ---
\begin{frame}{Analogy 5: ``the dealer secretly whispers a probability'' (= QMDP)}
QMDP rule: act as if the dealer privately whispered ``you're in Deck $s$ with probability $b(s)$,'' then take the action that maximises your expected $Q^\ast$:

\[
\boxed{\pi_{\rm QMDP}(\mathbf{b}) = \argmax_{\mathbf{a}} \sum_s b(s)\,Q^\ast(s, \mathbf{a})}
\]

\bigskip
\begin{itemize}
  \item Exact when $\mathbf{b}$ is a Dirac (fully observable limit).
  \item Cheap: reuses the MDP solver, no new optimisation per decision.
  \item Weakness: can't motivate information-gathering. Doesn't bind here, regimes are exogenous.
\end{itemize}
\end{frame}

% --- 11. Why POMDP not anything else ---
\begin{frame}{Why POMDP, not a tree or fully-observable MDP}
\centering
\small\renewcommand{\arraystretch}{1.25}
\begin{tabular}{ll}
\toprule
\textbf{Naive alternative} & \textbf{Why it fails here} \\
\midrule
Decision tree         & Explodes combinatorially with horizon \\
Influence diagram     & Same blow-up, not natural for repeated rebalance \\
Fully-observable MDP  & Requires knowing the regime, but it is \emph{latent} \\
\rowcolor{cb!10}\textbf{POMDP} & \textbf{Hidden state + noisy obs + sequential decision} \\
\bottomrule
\end{tabular}

\bigskip
\small In Powell (2019)'s 4-class taxonomy, QMDP is a \alert{value function approximation} (linear lower envelope over the belief simplex) + a \alert{one-step direct lookahead}.
\end{frame}

% --- 12. POMDP tuple ---
\begin{frame}{The POMDP tuple, written once}
\footnotesize
\begin{description}[itemsep=3pt,topsep=0pt,leftmargin=2cm]
  \item[$\mathcal{T}$] $=\{0,1,2,\dots\}$, monthly epochs.
  \item[$\state$] $=\{s_{\rm bull},\, s_{\rm bear}\}$ latent macro regimes.
  \item[$\action$] $=\{(0/100),(20/80),(40/60),(60/40),(80/20),(100/0)\}$.
  \item[$T(s'\mid s)$] HMM transition matrix; \alert{action-independent}.
  \item[$O(\mathbf{o}\mid s)$] $=\mathcal{N}(\boldsymbol\mu_s,\boldsymbol\Sigma_s)$ on $\mathbf{o}=(\mathrm{VIX},\mathrm{spread})$.
  \item[$R(s,\mathbf{a})$] $=\E[U(\mathbf{a}^\top\mathbf{r})\mid s]-c\lVert\mathbf{a}-\mathbf{a}_{\rm prev}\rVert_1$, CRRA $U$.
  \item[$\lambda$] $=0.95$ monthly discount.
\end{description}
\end{frame}

% --- 16. VI = PI ---
\begin{frame}{VI and PI agree exactly}
\small
The fully observable MDP satisfies
\[
V^\ast(s) = \max_{\mathbf{a}\in\action}\!\Bigl\{R(s,\mathbf{a}) + \lambda\!\sum_{s'\in\state}\!T(s'\mid s)\,V^\ast(s')\Bigr\}.
\]
\medskip
\begin{columns}[T]
\begin{column}{0.48\linewidth}
\textbf{Value iteration:} $V^{(n+1)}=LV^{(n)}$, stop at $\|\Delta V\|_\infty<2.6\times 10^{-6}$.

\smallskip
$\to$ \textbf{125 iters}.
\end{column}\begin{column}{0.48\linewidth}
\textbf{Policy iteration:} solve $(I-\lambda P_\pi)V=r_\pi$, greedy-improve.

\smallskip
$\to$ \textbf{2 iters}.
\end{column}
\end{columns}

\bigskip
\begin{tikzpicture}
\node[winner,text width=0.96\textwidth]{\centering
  \textbf{Proposition:} VI greedy policy $=$ PI greedy policy.\\
  Values agree within $2.6\times 10^{-6}$. Sanity check passed.
};
\end{tikzpicture}
\end{frame}

% --- 35. Extensions ---
\begin{frame}{Three extensions in priority order}
\bigskip
\begin{enumerate}\setlength{\itemsep}{12pt}
  \item \alert{CVaR-penalised reward}: replace CRRA with $R = \E[r] - \kappa\,\mathrm{CVaR}_\alpha[r]$ to explicitly penalise bear-regime tail risk. One-line change.

  \item \alert{Point-based value iteration (PBVI)}: tighter $\alpha$-vector bound on the 1-D belief simplex. Cheap here.

  \item \alert{Off-policy MC + Double Q-learning} (Precup 2000 + Van Hasselt 2010): a model-free baseline that doesn't assume the QMDP transition matrix.
\end{enumerate}
\end{frame}
