%-------- 1. Motivation --------
\begin{frame}{The problem: regimes break static allocation}
\textbf{In 2022 a 60/40 portfolio lost $\sim$17\%.}

Static allocation assumes stable joint return moments. Markets violate this in two ways:
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \alert{Conditional heteroskedasticity:} long quiet periods broken by short crises (2008, 2020, 2022).
  \item \alert{Stress correlation:} stocks and bonds sell off together when it hurts most. Bond diversification erodes precisely when it is most needed.
\end{itemize}

\bigskip
\begin{tikzpicture}[remember picture,overlay]
\node[draw=cb,thick,fill=cb!8,rounded corners=4pt,inner sep=8pt,text width=0.92\textwidth,anchor=south]
  at ([yshift=24mm]current page.south) {
    \textbf{Question.}\ Can a decision-theoretic regime overlay---a POMDP solved end-to-end on macroeconomic observables---beat a static benchmark after realistic frictions?
  };
\end{tikzpicture}
\end{frame}

%-------- 2. Why POMDP (rules out alternatives) --------
\begin{frame}{Why POMDP, not a tree, ID, or fully-observable MDP}
\centering
\small\renewcommand{\arraystretch}{1.25}
\begin{tabular}{ll}
\toprule
\textbf{Naive choice} & \textbf{Why it fails here} \\
\midrule
Decision tree         & Tree explodes combinatorially with horizon; no shared state. \\
Influence diagram     & Same blow-up; not natural for repeated monthly rebalance. \\
Fully-observable MDP  & Requires knowing the regime --- but it is \emph{latent}. \\
\rowcolor{cb!10}\textbf{POMDP} & \textbf{Hidden state, Bayesian belief, QMDP for tractable control.} \\
\bottomrule
\end{tabular}

\medskip
\textbf{Powell (2019) taxonomy:} QMDP is a \alert{value function approximation} (linear lower envelope over the belief simplex) combined with a \alert{one-step direct lookahead}. The right \emph{minimum} tool for our decision problem.
\end{frame}

%-------- 3. POMDP tuple --------
\begin{frame}{POMDP tuple}
\small
\begin{description}[itemsep=2pt,topsep=0pt]
  \item[$\mathcal{T}$] $=\{0,1,2,\dots\}$, monthly epochs.
  \item[$\state$] $=\{s_{\rm bull},s_{\rm bear}\}$ latent macro regimes ($K{=}2$).
  \item[$\action$] $=\{(0/100),(20/80),(40/60),(60/40),(80/20),(100/0)\}$ long-only weights.
  \item[$T(s'\mid s)$] HMM transition matrix; \alert{action-independent} (regimes exogenous).
  \item[$O(\mathbf{o}\mid s)$] Gaussian emission on $\mathbf{o}=(\mathrm{VIX},\mathrm{term\ spread})$.
  \item[$R(s,\mathbf{a})$] $=\E[U(\mathbf{a}^\top\mathbf{r})\mid s]-c\lVert\mathbf{a}-\mathbf{a}_{\rm prev}\rVert_1$,\ CRRA $U$.
  \item[$\lambda=0.95$] monthly discount.
\end{description}
\medskip
\textbf{Bayesian filter:}
\[
  \bel_t(s')\propto O(\mathbf{o}_t\mid s')\!\sum_s T(s'\mid s)\,\bel_{t-1}(s).
\]
\end{frame}

%-------- 6. VI vs PI --------
\begin{frame}{Underlying MDP: VI and PI agree exactly}
\small
The fully observable MDP satisfies the Bellman optimality equation
\[
V^\ast(s)=\max_{\mathbf{a}\in\action}\!\Bigl\{R(s,\mathbf{a})+\lambda\!\sum_{s'\in\state}\!T(s'\mid s)\,V^\ast(s')\Bigr\}.
\]
\medskip
\begin{columns}[T]
\begin{column}{0.48\linewidth}
\textbf{Value iteration:}\\
$V^{(n+1)}=LV^{(n)}$, stop at $\lVert\Delta V\rVert_\infty<\varepsilon(1-\lambda)/(2\lambda)$.\\[2pt]
$\to$ \textbf{125 iters} for $\lambda{=}0.95$, $\varepsilon{=}10^{-4}$.
\end{column}\begin{column}{0.48\linewidth}
\textbf{Policy iteration:}\\
solve $(I-\lambda P_\pi)V=r_\pi$, then greedy-improve.\\[2pt]
$\to$ \textbf{2 iters} to converge.
\end{column}
\end{columns}
\bigskip
\begin{block}{Proposition (numerical solver agreement)}
\small VI greedy policy $=$ PI greedy policy. Value functions agree within $2.6\times 10^{-6}$. Empirical realisation of the contraction-mapping equivalence (Puterman 2005).
\end{block}
\end{frame}

%-------- 7. QMDP rule --------
\begin{frame}{QMDP: project belief onto $Q^\ast$}
\[
\pi_{\rm QMDP}(\bel)=\argmax_{\mathbf{a}\in\action}\;\sum_{s\in\state}\bel(s)\,Q^\ast(s,\mathbf{a})
\]

\textbf{Worked example.} Belief after a high-VIX observation: $\bel=(0.069,\,0.931)$ (post-VIX-35 update).

\smallskip
At $\gamma{=}8$ where the policy actually differs across regimes:

\centering\small\renewcommand{\arraystretch}{1.1}
\begin{tabular}{l rr r}
\toprule
Action $\mathbf{a}$ & $Q^\ast(\text{bull})$ & $Q^\ast(\text{bear})$ & $\bel^\top Q^\ast(\cdot,\mathbf{a})$ \\
\midrule
100/0 & 0.32 & 0.10 & 0.115 \\
60/40 & 0.24 & 0.20 & 0.203 \\
\rowcolor{cg!18}\textbf{40/60} & 0.20 & \textbf{0.22} & \textbf{0.219} \\
0/100 & 0.05 & 0.18 & 0.171 \\
\bottomrule
\end{tabular}

\medskip
\small QMDP picks 40/60. \alert{But at our headline $\gamma{=}2$, $Q^\ast$ has ``100/0'' dominating both columns} $\Rightarrow$ QMDP always picks 100/0. The collapse.
\end{frame}

%-------- 9. Why QMDP collapsed: gamma table --------
\begin{frame}{Why QMDP collapsed: CRRA $\gamma$ never separates regimes}
\centering\small\setlength{\tabcolsep}{8pt}\renewcommand{\arraystretch}{1.1}
\begin{tabular}{lccccccc}
\toprule
\textbf{Regime} & $\gamma{=}1$ & $\gamma{=}2$ & $\gamma{=}3$ & $\gamma{=}5$ & $\gamma{=}8$ & $\gamma{=}15$ & $\gamma{=}20$ \\
\midrule
Bull (0) & 100/0 & 100/0 & 100/0 & 80/20 & 40/60 & 20/80 & 20/80 \\
Bear (1) & 100/0 & 100/0 & 100/0 & 80/20 & 40/60 & 20/80 & 20/80 \\
\bottomrule
\end{tabular}

\bigskip
\small
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \alert{Within columns:} bull and bear give the \emph{same} action. The regime label never moves the policy.
  \item \alert{Across columns:} higher $\gamma$ shifts the policy bondward across-the-board.
  \item The Sharpe gain from raising $\gamma$ is a \emph{second-moment effect} (less equity $\to$ less vol $\to$ higher Sharpe), \textbf{not} regime-aware tilting.
\end{itemize}
\end{frame}

\begin{frame}{Unlock 1: richer observations \emph{fix} the policy collapse}

\begin{tabular}{l l cc c}
\toprule
\textbf{Cohort} & \textbf{Features} & \textbf{Bull} & \textbf{Bear} & \textbf{Differs?} \\
\midrule
1. Baseline         & VIX, T10Y3M                & 100/0 & 100/0 & \ding{55} \\
2. +Yield curve     & +T10Y2Y                    & 100/0 & 100/0 & \ding{55} \\
\rowcolor{cg!18}\textbf{3. +Stress} & \textbf{+NFCI, +STLFSI4}   & \textbf{100/0} & \textbf{0/100} & \textbf{\ding{51}\ full flip} \\
4. +Macro           & +NFCI, +UMCSENT, +ICSA     & 100/0 & 0/100 & \ding{51} \\
5. Kitchen sink     & 8 channels                 & 100/0 & 60/40 & \ding{51} \\
\bottomrule

\medskip
\small Same $\gamma{=}2$ as headline. Adding the Chicago Fed NFCI and St.\ Louis Fed STLFSI4 (the financial-stress indicators that substitute for the FRED-truncated HY OAS) makes the regime policy \emph{fully differentiate}: stocks in bull, bonds in bear.

\medskip
\alert{The headline collapse is an observation-channel problem, not a fundamental QMDP problem.} Confirms Guidolin--Timmermann (2007) prediction.
\end{frame}

%-------- 11. UNLOCK 2: walk-forward refit --------
\begin{frame}{Unlock 2: walk-forward refit lifts QMDP Sharpe to 1.08}
\centering\small\setlength{\tabcolsep}{7pt}\renewcommand{\arraystretch}{1.1}
\begin{tabular}{l rrrr}
\toprule
\textbf{Variant} & \textbf{CAGR} & \textbf{Sharpe} & \textbf{Max DD} & \textbf{Calmar} \\
\midrule
Static 60/40 (bench)         & 7.40\% & 0.81 & $-34.2\%$ & 0.22 \\
QMDP fixed (report baseline) & 9.87\% & 0.96 & $-34.2\%$ & 0.29 \\
QMDP annual refit (5y win)   & 9.11\% & 0.85 & $-25.1\%$ & 0.36 \\
QMDP quarterly refit         & 8.56\% & 0.83 & $-23.4\%$ & 0.37 \\
\rowcolor{cg!18}\textbf{QMDP expanding window} & 9.81\% & \textbf{1.08} & $\mathbf{-16.5\%}$ & \textbf{0.60} \\
HMM-MV expanding window      & 5.32\% & 0.95 & $-16.8\%$ & 0.32 \\
\bottomrule
\end{tabular}

\medskip
\small Following Nystrup--Madsen--Lindstr\"om (2018): expanding-window walk-forward refit
\begin{itemize}\setlength{\itemsep}{1pt}
  \item lifts QMDP Sharpe \alert{$0.81 \to 1.08$} (+33\%);
  \item cuts max drawdown \alert{$-34\% \to -16.5\%$} (more than halves);
  \item nearly triples Calmar \alert{$0.22 \to 0.60$}.
\end{itemize}

\medskip
\alert{The fixed-window HMM was a methodology choice masking real performance.}
\end{frame}

%-------- 13. Conclusions + extensions --------
\begin{frame}{Conclusions \& extensions}
\textbf{Three findings.}
\begin{enumerate}\setlength{\itemsep}{2pt}
  \item Practitioner-consensus rule (Faber 10-mo SMA) dominates 12-strategy horse race; QMDP at $\gamma{=}2$ comes last.
  \item Adding NFCI + STLFSI4 \emph{fixes} the policy collapse at the same $\gamma{=}2$.
  \item Expanding-window walk-forward refit lifts QMDP Sharpe to 1.08 and halves max drawdown.
\end{enumerate}

\medskip
\textbf{Re-framing.} Our headline negative finding holds \emph{only} in the joint configuration of (a) narrow observations, (b) fixed HMM, (c) low CRRA. Each component is empirically reversible. \alert{Diagnostic of methodology, not a fundamental verdict.}

\medskip
\textbf{Extensions in priority order.}
\begin{enumerate}\setlength{\itemsep}{1pt}
  \item \alert{CVaR-penalised reward:} explicitly penalise bear-regime tail risk.
  \item \alert{Point-based VI:} tighter $\alpha$-vector bound on a 1-D belief simplex (cheap).
  \item \alert{Off-policy MC + Double Q-learning} (Precup 2000 + Van Hasselt 2010) as model-free baseline.
\end{enumerate}
\end{frame}
